  const DATA = {
    // Hero banner image (you already have this file in repo root)
    HERO_IMAGE: "./image_hero.jpg",

    // Two products for the shop
    PRODUCTS: [
      {
        id: "toner",                    // short key, no spaces
        title: "Hydrating Toner",       // name shown on the card
        price: 24.00,                   // numeric price
        image: "./hydrating-toner.jpg", // file in repo root
        size: "50 ml",                  // optional subtitle
        tag: "Hydration"                // optional short label
      },
      {
        id: "serum",
        title: "Vitamin C Serum",
        price: 28.00,
        image: "./serum.jpg",
        size: "30 ml",
        tag: "Brighten"
      }
    ]

  <!-- SITE LOGIC (no need to edit) -->
  <script>
    const $ = (s) => document.querySelector(s);
    const fmt = (n) => `$${n.toFixed(2)}`;

    // Show hero image
    $('#heroImage').src = DATA.HERO_IMAGE;

    // Render product cards into the grid
    const grid = $('#productGrid');
    function renderProducts(){
      grid.innerHTML = DATA.PRODUCTS.map(p => `
        <article class="card">
          <img alt="${p.title}" src="${p.image}">
          <div class="info">
            <div style="font-weight:700">${p.title}</div>
            <div class="muted">${p.size || ""} ${p.tag ? `• ${p.tag}` : ""}</div>
            <div class="price">
              <span>${fmt(p.price)}</span>
              <button class="btn btn-ghost" onclick="addToCart('${p.id}')">Add to cart</button>
            </div>
          </div>
        </article>
      `).join('');
    }

    // Simple in-memory cart
    const cart = {}; // { id: qty }
    function addToCart(id){ cart[id] = (cart[id]||0) + 1; updateCartUI(); openDrawer(); }
    function setQty(id, qty){ cart[id] = Math.max(1, parseInt(qty)||1); updateCartUI(); }
    function removeFromCart(id){ delete cart[id]; updateCartUI(); }

    function cartItems(){
      return Object.entries(cart).map(([id,qty])=>{
        const p = DATA.PRODUCTS.find(x=>x.id===id); return p ? {...p, qty} : null;
      }).filter(Boolean);
    }
    function totals(){
      const items = cartItems();
      const subtotal = items.reduce((s,i)=> s + i.price*i.qty, 0);
      const tax = +(subtotal*0.08).toFixed(2);              // demo: 8% tax
      const shipping = subtotal>50 || !items.length ? 0 : 6.5; // free > $50
      return {subtotal, tax, shipping, total: subtotal+tax+shipping};
    }
    function updateBadge(){
      const count = Object.values(cart).reduce((a,b)=>a+b,0);
      $('#cartCount').textContent = count;
    }
    function updateCartUI(){
      updateBadge();
      const items = cartItems();
      const list = document.getElementById('cartItems');
      if(!items.length){ list.innerHTML = '<div class="muted">Your cart is empty.</div>'; }
      else {
        list.innerHTML = items.map(i=>`
          <div class="row">
            <img alt="${i.title}" src="${i.image}">
            <div>
              <strong>${i.title}</strong>
              <div class="muted">${fmt(i.price)}</div>
              <div class="qty">
                <button onclick="setQty('${i.id}', ${i.qty-1})">−</button>
                <input value="${i.qty}" oninput="setQty('${i.id}', this.value)">
                <button onclick="setQty('${i.id}', ${i.qty+1})">+</button>
              </div>
            </div>
            <div>
              <div>${fmt(i.qty*i.price)}</div>
              <button class="btn btn-ghost" onclick="removeFromCart('${i.id}')">Remove</button>
            </div>
          </div>
        `).join('');
      }
      const t = totals();
      document.getElementById('subtotal').textContent = fmt(t.subtotal);
      document.getElementById('tax').textContent = fmt(t.tax);
      document.getElementById('shipping').textContent = fmt(t.shipping);
      document.getElementById('total').textContent = fmt(t.total);
    }

    // Drawer controls
    const drawer = document.getElementById('cartDrawer');
    function openDrawer(){ drawer.classList.add('open'); }
    function closeDrawer(){ drawer.classList.remove('open'); }
    document.getElementById('openCart').addEventListener('click', e=>{ e.preventDefault(); openDrawer(); updateCartUI(); });
    document.getElementById('viewCartBtn')?.addEventListener('click', ()=>{ openDrawer(); updateCartUI(); });
    document.getElementById('closeCart').addEventListener('click', closeDrawer);

    // Init
    renderProducts();
    updateCartUI();
  </script>
